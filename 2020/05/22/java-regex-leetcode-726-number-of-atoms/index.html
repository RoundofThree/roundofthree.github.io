Java regex: Leetcode 726. Number of Atoms

recursion,
Publish Date:   2020-05-22
Update Date:   2020-05-22
Word Count:   520
Read Times:   3 Min

This challenge asks to parse a chemical formula. A parsing problem immediately makes me think of regular expressions. They are really helpful! All defined patterns are grouped in order of appearance. Isn’t it marvelous? Let’s compare a hand parsing method and a regex method. The logic is easy, so what makes this problem hard is that there are a handful of different patterns to look for. But it is a lot easier if you master the java.util.regex package!

Recursive method and hand parsing

class Solution {
    public String countOfAtoms(String formula) {
        TreeMap<String, Integer> map = getMap(formula);
        // structure the return string
        StringBuilder sb = new StringBuilder();
        for(String atom : map.keySet()) {
            sb.append(atom).append(map.get(atom)>1? map.get(atom)+"":"");
        }
        return sb.toString();
    }

    private int findClosedParenthesis(String s, int start) {
        int count = 1;
        for(int i=start+1; i<s.length(); i++) {
            if(s.charAt(i) == '(') count++;
            else if(s.charAt(i) == ')') count--;
            if(count == 0) {
                return i;
            }
        }
        return s.charAt(s.length()-1);  
    }

    private TreeMap getMap(String formula) {
        TreeMap<String, Integer> map = new TreeMap<>();
        if(formula.isEmpty()) return map;

        int i=0;
        while(i < formula.length()) {
            // starting an embedded formula, so apply recursion
            if(formula.charAt(i) == '(') {
                int j = findClosedParenthesis(formula, i);
                Map<String, Integer> submap = getMap(formula.substring(i+1,j));
                j++;
                // find the multiple of the embedded formula
                int k = j;
                while(k < formula.length() && Character.isDigit(formula.charAt(k))) k++;
                int mul = k > j? Integer.parseInt(formula.substring(j,k)) : 1;
                // add it to map
                for(String atom : submap.keySet()) {
                    map.put(atom, submap.get(atom)*mul + map.getOrDefault(atom, 0));
                }
                i = k;

            } else {
                int j = i + 1;
                while(j <formula.length() && formula.charAt(j)>='a' && formula.charAt(j)<='z') j++;
                String atom = formula.substring(i,j);

                int k = j;
                while(k < formula.length() && Character.isDigit(formula.charAt(k))) k++;
                int mul = k > j? Integer.parseInt(formula.substring(j, k)) : 1;
                // update map
                map.put(atom, mul + map.getOrDefault(atom, 0));
                i=k;
            }
        }
        return map;
    }
}

Regex

import java.util.regex.*;

class Solution {
    public String countOfAtoms(String formula) {
        Matcher matcher = Pattern.compile("([A-Z][a-z]*)(\\d*)|(\\()|(\\))(\\d*)").matcher(formula);
        Stack<Map<String, Integer>> stack = new Stack();
        stack.push(new TreeMap());

        while (matcher.find()) {
            String match = matcher.group();
            if (match.equals("(")) {
                stack.push(new TreeMap());
            } else if (match.startsWith(")")) {
                Map<String, Integer> top = stack.pop();
                int multiplicity = match.length() > 1 ? Integer.parseInt(match.substring(1, match.length())) : 1;
                for (String name: top.keySet()) {
                    stack.peek().put(name, stack.peek().getOrDefault(name, 0) + top.get(name) * multiplicity);
                }
            } else {
                int i = 1;
                while (i < match.length() && Character.isLowerCase(match.charAt(i))) {
                    i++;
                }
                String name = match.substring(0, i);
                int multiplicity = i < match.length() ? Integer.parseInt(match.substring(i, match.length())) : 1;
                stack.peek().put(name, stack.peek().getOrDefault(name, 0) + multiplicity);
            }
        }

        StringBuilder ans = new StringBuilder();
        for (String name: stack.peek().keySet()) {
            ans.append(name);
            final int count = stack.peek().get(name);
            if (count > 1) ans.append(String.valueOf(count));
        }
        return ans.toString();
    }
}
